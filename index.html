function doGet(e) { var action = e.parameter.action; // معرفات الشيتات الخاصة بك (IDs) var dbSourceId = "1HTAXntyJ_5RfEi5rxcgwUs3mGdAKda_5-ODqFLqHTi4"; // شيت المستخدمين والمواد var ordersDbId = "15e_uxz71uj8Co-_RFb_9KWjO-xSX6XdL90nX2UMBmJE"; // شيت الطلبات // 1. التحقق من الإيميل والباسوورد وجلب الاسم من شيت Users if (action === "verifyUser") { var emailToVerify = e.parameter.email; var passwordToVerify = e.parameter.password; var ssSource = SpreadsheetApp.openById(dbSourceId); var userSheet = ssSource.getSheetByName("Users"); var authorized = false; var userName = ""; if (userSheet) { var data = userSheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { var sheetEmail = data[i][0].toString().trim().toLowerCase(); var sheetPassword = data[i][1].toString().trim(); var sheetUserName = data[i][2] ? data[i][2].toString().trim() : ""; if (sheetEmail === emailToVerify.trim().toLowerCase() && sheetPassword === passwordToVerify) { authorized = true; userName = sheetUserName || sheetEmail; break; } } } return ContentService.createTextOutput(JSON.stringify({ authorized: authorized, userName: userName })) .setMimeType(ContentService.MimeType.JSON); } // 2. جلب المواد من شيت Materials if (action === "getMaterials") { var ssSource = SpreadsheetApp.openById(dbSourceId); var sheet = ssSource.getSheetByName("Materials"); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Materials sheet not found." })) .setMimeType(ContentService.MimeType.JSON); } var data = sheet.getDataRange().getValues(); var headers = data[0]; var jsonArray = []; for (var i = 1; i < data.length; i++) { var obj = {}; for (var j = 0; j < headers.length; j++) { obj[headers[j]] = data[i][j]; } jsonArray.push(obj); } return ContentService.createTextOutput(JSON.stringify({ success: true, data: jsonArray })) .setMimeType(ContentService.MimeType.JSON); } // 3. إرسال الطلب وحفظه في الشيت الثاني (Orders) if (action === "submitOrder") { try { var project = e.parameter.project; var location = e.parameter.location; var requester = e.parameter.requester; var orders = JSON.parse(e.parameter.orders); var ssOrders = SpreadsheetApp.openById(ordersDbId); var orderSheet = ssOrders.getSheetByName("Orders"); if (!orderSheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Orders sheet not found in destination." })) .setMimeType(ContentService.MimeType.JSON); } var timestamp = new Date(); var prNo = "PR-" + timestamp.getFullYear() + "-" + Math.floor(Math.random() * 9000 + 1000); orders.forEach(function(item) { orderSheet.appendRow([ prNo, timestamp, project, location, requester, item.code, item.description, item.qty ]); }); return ContentService.createTextOutput(JSON.stringify({ success: true, prNo: prNo })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ success: false, message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } }
